#creating 2 or more plots in a single fig

#customization

plt.title("Graph 1")

plt.title("Graph 2")

#Line plot with customization
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
a=np.array([10,20,30,40,50,60,70,80]) #btn 0 and 10 with interval of 0.1
b=np.array([5,3,2,5,1,6,7,4])
c=np.array([1,2,3,4,6,1,2,5])

#customization
fig=plt.figure(figsize=(8,5))
plt.scatter(a,b,color="blue",s=200,edgecolors='r',alpha=0.7,marker='o')
plt.scatter(a,c,color="red",s=400,edgecolors='b',alpha=0.7,marker='4')
plt.title("Scatter plot demo")
plt.xlabel("x-axis")
plt.ylabel("y-axis")
plt.legend(['b','c'],loc='best')
plt.grid(True)
plt.savefig('scatterplot.png')
plt.show()
